ALPINE COMPONENT (clone the 04-RESEARCH feedPage() at lines 627-654 + connections.html `_act(busy)` pattern):
```js
function feedPage() {
  return {
    posts: [], loading: true, error: '', busy: false, degraded: false,
    draft: { content: '', image_url: '' }, openComments: {}, commentDraft: {}, comments: {},
    reactions: ['like','love','haha','wow','sad','angry'],
    async load() {
      try { const r = await api.get('/feed?limit=20'); this.posts = r.data; this.degraded = !!(r.meta && r.meta.degraded); }
      catch (e) { this.error = e.message; } finally { this.loading = false; }
    },
    async submit() {
      if (this.busy || !this.draft.content.trim()) return; this.busy = true;
      try { await api.post('/posts', { content: this.draft.content, image_url: this.draft.image_url || undefined });
            this.draft = { content: '', image_url: '' }; await this.load(); }
      catch (e) { this.error = e.message; } finally { this.busy = false; }
    },
    react(id, type)   { return this._act(() => api.post('/posts/' + id + '/reactions', { type })); },
    unreact(id)       { return this._act(() => api.delete('/posts/' + id + '/reactions')); },
    repost(id)        { return this._act(() => api.post('/posts/' + id + '/repost')); },
    removePost(id)    { return this._act(() => api.delete('/posts/' + id)); },
    async loadComments(id) { const r = await api.get('/posts/' + id + '/comments'); this.comments[id] = r.data; this.openComments[id] = true; },
    async addComment(id) { const b = (this.commentDraft[id]||'').trim(); if(!b) return; await api.post('/posts/'+id+'/comments',{body:b}); this.commentDraft[id]=''; await this.loadComments(id); await this.load(); },
    async deleteComment(pid, cid) { await api.delete('/comments/'+cid); await this.loadComments(pid); await this.load(); },
    async _act(fn) { if (this.busy) return; this.busy = true; try { await fn(); await this.load(); } catch(e){ this.error=e.message; } finally { this.busy = false; } },
  };
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user content → DOM | post content, comment body, author names rendered in the browser |
| browser → gateway | window.api attaches the Bearer JWT; all writes are gateway-authenticated |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-19 | Tampering/Info disclosure (stored XSS) | feed.html rendering | mitigate | `x-text` for all user-supplied content (content/comment body/author names/original content); NEVER `x-html` (verify grep) — D-10. |
| T-04-20 | Info disclosure (PII) | feed/comment cards | mitigate | Cards come from the gateway, which already allowlists email out (Plan 03); the human-verify step confirms no email appears. |
| T-04-21 | Elevation (delete others' content) | delete buttons | mitigate | Delete button shown only when `author_id === auth.user.id`; the gateway+feed-service enforce owner-scope server-side regardless of the UI. |
</threat_model>
