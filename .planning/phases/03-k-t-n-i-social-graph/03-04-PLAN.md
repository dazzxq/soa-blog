    api.get('/connections'), api.get('/connections/requests?direction=incoming'),

    assign .data to each. try/catch sets this.error (Vietnamese). finally loading=false.
  async sendInvite(targetId) { await api.post('/connections/requests', {target_id:targetId}); await this.load(); }
  async accept(reqId) { await api.post('/connections/requests/'+reqId+'/accept'); await this.load(); }
  async reject(reqId) { await api.post('/connections/requests/'+reqId+'/reject'); await this.load(); }
  async cancel(reqId) { await api.delete('/connections/requests/'+reqId); await this.load(); }
  async remove(userId) { await api.delete('/connections/'+userId); await this.load(); }
  Guard each action with this.busy to prevent double-submit; surface errors (e.message) via this.error.
